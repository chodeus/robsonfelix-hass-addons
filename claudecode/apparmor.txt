#include <tunables/global>

profile claudecode flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities needed for terminal and networking
  capability net_admin,
  capability net_raw,
  capability setgid,
  capability setuid,
  capability chown,
  capability fowner,
  capability dac_override,
  capability dac_read_search,
  capability sys_tty_config,

  # Network access for Claude API and ha-mcp
  network,

  # PTY/TTY access for terminal
  /dev/ptmx rw,
  /dev/pts/* rw,
  /dev/tty* rw,
  /dev/null rw,
  /dev/zero r,
  /dev/urandom r,

  # System files (read-only)
  /etc/** r,
  /usr/** r,
  /lib/** r,
  /proc/** r,
  /sys/** r,

  # Execute permissions for binaries
  /bin/** ixr,
  /usr/bin/** ixr,
  /usr/lib/** ixmr,
  # ixmr (not ixr): the base image builds CPython + native .so extensions (ha-mcp's
  # pydantic-core, cryptography) under /usr/local, which need mmap(PROT_EXEC) to load.
  /usr/local/** ixmr,
  /usr/libexec/** ixr,

  # iptables lives in /sbin on Alpine; needed for the opt-in restrict_terminal_port firewall.
  /sbin/** ixmr,

  # Docker socket — config.yaml sets docker_api: true and the image ships docker-cli;
  # without this rule every `docker` command fails to connect (Alpine's /var/run -> /run).
  /run/docker.sock rw,
  # iptables (nft or legacy backend) takes this lock; without it `iptables -A` fails.
  /run/xtables.lock rwk,

  # Add-on data (private, non-backed-up volume)
  /data/ r,
  /data/** rwk,

  # Home Assistant config - main working directory
  /homeassistant/ r,
  # `l` (link): Claude Code's postinstall hardlinks its ~256MB native binary out of the
  # platform package. Without link permission it falls back to a full copy, doubling what
  # every HA backup carries.
  /homeassistant/** rwkl,
  # Allow executing Claude Code installed via in-container update (npm --prefix)
  /homeassistant/.claudecode/npm-global/** ixmr,

  # Other mapped HA directories
  /ssl/ r,
  /ssl/** r,
  /share/ r,
  /share/** rwk,
  /media/ r,
  /media/** rwk,
  /backup/ r,
  /backup/** r,
  /addon_configs/ r,
  /addon_configs/** rwk,
  /config/ r,
  /config/** rwk,

  # Claude Code and user home
  /root/ r,
  /root/** rwk,
  /tmp/** rwk,

  # Node.js modules
  /usr/lib/node_modules/** ixmr,
}
